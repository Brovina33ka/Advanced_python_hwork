def tenet(input_file_path: str, output_file_path: str) -> None:
    if ".txt" not in input_file_path:
        raise ValueError(" Вместо файла получена директория")
    elif input_file_path == output_file_path:
        raise ValueError(" Введите другой файл вывода")
    else:
        with open(input_file_path, "r", encoding="utf8") as input_f, open(output_file_path, "w") as output:
            for line in input_f:
                output.write(line[::-1])
        input_f.close()
        output.close()
